# 1. Tạo conda environment:
#    conda create -n dualmamba python=3.10 && conda activate dualmamba
#
# 2. Cài PyTorch (chọn đúng CUDA version):
#    pip install torch==2.1.0 --index-url https://download.pytorch.org/whl/cu118   # CUDA 11.8
#    pip install torch==2.1.0 --index-url https://download.pytorch.org/whl/cu121   # CUDA 12.1
#
# 3. Cài mamba-ssm (cần CUDA, cài sau torch):
#    pip install causal-conv1d>=1.1.0
#    pip install mamba-ssm
#    # Nếu gặp lỗi compile: pip install mamba-ssm --no-build-isolation
#
# 4. Cài DGL:
#    pip install dgl -f https://data.dgl.ai/wheels/repo.html   # CUDA
